} function addItem(payload) { // payload: itemName, category, unit ("g" | "ml" | "pcs"), costPerUnit (optional, defaults 0) const items = readTable("Items"); const existing = items.find(i => String(i.ItemName).toLowerCase() === String(payload.itemName).toLowerCase()); } function createSingleReportRow(payload) { // payload: outlet, itemName, category, unit ("g"|"ml"|"pcs"), qty, posterName, // notes, photoBase64, photoMimeType const items = getItems(); let item = items.find(i => String(i.ItemName).toLowerCase() === String(payload.itemName).toLowerCase()); } const unit = item.Unit || payload.unit || "g"; const costPerUnit = Number(item.CostPerUnit) || 0; const qty = Number(payload.qty) || 0; const totalCost = qty * costPerUnit; let photoUrl = ""; if (payload.photoBase64) { photoUrl = savePhoto(payload.photoBase64, payload.photoMimeType || "image/jpeg"); } // Auto-register outlet if it's a new one typed by staff const outlets = getOutlets(); if (!outlets.find(o => o.OutletName === payload.outlet)) { addOutlet({ name: payload.outlet }); } const reportId = "RPT-" + Date.now() + "-" + Math.floor(Math.random() * 1000); appendRow("Reports", { ReportID: reportId, Timestamp: new Date(), Outlet: payload.outlet, ItemName: item.ItemName, Category: payload.category || item.Category || "", Qty: qty, Unit: unit, UnitCost: costPerUnit, TotalCost: Number(totalCost.toFixed(2)), PosterName: payload.posterName, PhotoURL: photoUrl, Notes: payload.notes || "" }, ["ReportID", "Timestamp", "Outlet", "ItemName", "Category", "Qty", "Unit", "UnitCost", "TotalCost", "PosterName", "PhotoURL", "Notes"]); return { reportId, totalCost: Number(totalCost.toFixed(2)), unit, isNewItem: !Number(item.CostPerUnit) }; } function savePhoto(base64Data, mimeType) { const folders = DriveApp.getFoldersByName(PHOTO_FOLDER_NAME); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return "https://drive.google.com/uc?id=" + file.getId(); } /* ---------------- AUTH ---------------- */ function supervisorLogin(payload) { const rows = readTable("Supervisors"); } /* ---------------- INVENTORY ---------------- */ // InventoryReceipts sheet stores ONLY manual "Item Received" entries. // Wasted is always pulled live from Reports. Available/Closing stock is // computed by rolling forward day by day, so it self-corrects even if a // past day's data changes later — no stale numbers, no manual carry-forward. function toDateStr(d) { const dt = (d instanceof Date) ? d : new Date(d); return Utilities.formatDate(dt, Session.getScriptTimeZone(), "yyyy-MM-dd"); } function dayName(dateStr) { return Utilities.formatDate(new Date(dateStr), Session.getScriptTimeZone(), "EEEE"); } function getInventory(payload) { // payload: { date: "yyyy-MM-dd", outlet } const date = payload.date; const reports = readTable("Reports").filter(r => r.Outlet === outlet); const overrides = readTable("InventoryOpeningOverrides").filter(r => r.Outlet === outlet); const allDatesSet = new Set(); receipts.forEach(r => allDatesSet.add(toDateStr(r.Date))); reports.forEach(r => allDatesSet.add(toDateStr(r.Timestamp))); overrides.forEach(r => allDatesSet.add(toDateStr(r.Date))); allDatesSet.add(date); const allDates = Array.from(allDatesSet).filter(d => d <= date).sort(); const wasted = reports .filter(r => toDateStr(r.Timestamp) === d && r.ItemName === item.ItemName) .reduce((s, r) => s + Number(r.Qty || 0), 0); const closing = opening + received - wasted; if (d === date) { rowForDate = { Date: date, Day: dayName(date), ItemName: item.ItemName, Unit: item.Unit, Opening: opening, Received: received, Wasted: wasted, Available: closing, CostPerUnit: Number(item.CostPerUnit) || 0, Value: Number(((opening + received - wasted) * (Number(item.CostPerUnit) || 0)).toFixed(2)) }; } opening = closing; }); return rowForDate || { Date: date, Day: dayName(date), ItemName: item.ItemName, Unit: item.Unit, Opening: 0, Received: 0, Wasted: 0, Available: 0, CostPerUnit: Number(item.CostPerUnit) || 0, Value: 0 }; }); } function setOpeningStock(payload) { // payload: date, outlet, itemName, qty — sets/resets the true physical count as of this date const sh = sheet("InventoryOpeningOverrides"); const data = sh.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (toDateStr(data[r][dateCol]) === payload.date && data[r][outletCol] === payload.outlet && data[r][itemCol] === payload.itemName) { sh.getRange(r + 1, openCol + 1).setValue(Number(payload.qty) || 0); return { updated: true }; } } appendRow("InventoryOpeningOverrides", { Date: payload.date, Outlet: payload.outlet, ItemName: payload.itemName, OpeningStock: Number(payload.qty) || 0 }, ["Date", "Outlet", "ItemName", "OpeningStock"]); return { added: true }; } function setReceived(payload) { // payload: date, outlet, itemName, qty const sh = sheet("InventoryReceipts"); const data = sh.getDataRange().getValues(); const headers = data[0]; r++) { if (toDateStr(data[r][dateCol]) === payload.date && data[r][outletCol] === payload.outlet && data[r][itemCol] === payload.itemName) { sh.getRange(r + 1, recvCol + 1).setValue(Number(payload.qty) || 0); return { added: true }; } /* ---------------- REPORT QUERIES ---------------- */ function getReports(payload) { // payload: { scopeOutlet (optional, restricts to one outlet for supervisors), // outlet, itemName, category, dateFrom, dateTo } let rows = readTable("Reports"); if (payload.scopeOutlet) { rows = rows.filter(r => r.Outlet === payload.scopeOutlet); } if (payload.outlet) { rows = rows.filter(r => r.Outlet === payload.outlet); } if (payload.itemName) { rows = rows.filter(r => r.ItemName === payload.itemName); } if (payload.category) { rows = rows.filter(r => r.Category === payload.category); } if (payload.dateFrom) { const from = new Date(payload.dateFrom); rows = rows.filter(r => new Date(r.Timestamp) >= from); } if (payload.dateTo) { const to = new Date(payload.dateTo); to.setHours(23, 59, 59, 999); rows = rows.filter(r => new Date(r.Timestamp) <= to); } rows.sort((a, b) => new Date(b.Timestamp) - new Date(a.Timestamp)); return rows; }
